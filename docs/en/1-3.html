Topic 01

NavigationStack + NavigationPath

Stack-based navigation and programmatic control
Swift 
struct RootView: View {
    @State private var path = NavigationPath()

    var body: some View {
        NavigationStack(path: $path) {
            List {
                NavigationLink("Profile",
                    value: "profile")
                NavigationLink("Settings",
                    value: "settings")
            }
            .navigationTitle("Menu")
            .navigationDestination(
                for: String.self
            ) { value in
                Text("\(value) screen")
            }
        }
    }
}
Think About It
  • What is the difference between NavigationLink(destination:) and .navigationDestination(for:)?
  • What becomes possible when you manage NavigationPath with @State?
  • How do you manipulate NavigationPath to navigate directly to a specific screen via code?
NavigationLink(destination:)
Pushes a screen on tap. Back navigation is automatic. .navigationDestination(for:) pattern (iOS 16+)
NavigationPath
Manage path with @State to navigate programmatically. path.append() / path.removeLast() / path.removeAll()
Deep Link Handling
Manipulate NavigationPath in onOpenURL to navigate directly to a specific screen
Topic 02

sheet vs fullScreenCover

Modal types and when to choose each
Swift 
struct ModalDemo: View {
    @State private var showSheet = false
    @State private var showFull = false

    var body: some View {
        VStack(spacing: 16) {
            Button("Open Sheet") {
                showSheet = true
            }
            Button("Open FullScreen") {
                showFull = true
            }
        }
        .sheet(isPresented: $showSheet) {
            Text("Sheet!")
                .presentationDetents(
                    [.medium, .large])
        }
        .fullScreenCover(
            isPresented: $showFull
        ) {
            Text("Full Screen!")
        }
    }
}
Think About It
  • How do .sheet() and .fullScreenCover() differ visually?
  • What environment variable do you use to dismiss a sheet from within?
  • What does .presentationDetents([.medium, .large]) control?
.sheet(isPresented: $show)
A card-style modal sliding up from the bottom. The previous screen is partially visible. The most common modal type
.fullScreenCover(isPresented: $show)
Covers the entire screen. Used for camera, onboarding, or any completely new context
dismiss environment variable
@Environment(\.dismiss) var dismiss — dismiss the modal from a child View
Topic 03

@StateObject vs @ObservedObject

Ownership vs subscription
Swift 
class TodoStore: ObservableObject {
    @Published var items: [String] = []

    func add(_ item: String) {
        items.append(item)
    }
}

struct ParentView: View {
    @StateObject var store = TodoStore()

    var body: some View {
        ChildView(store: store)
    }
}

struct ChildView: View {
    @ObservedObject var store: TodoStore

    var body: some View {
        List(store.items, id: \.self) {
            Text($0)
        }
    }
}
Think About It
  • When should you use @StateObject vs @ObservedObject?
  • Why is it dangerous to use @StateObject in a child View?
  • What happens when a @Published property of an ObservableObject changes?
  • Why use a class? Why can't you use a struct instead?
ObservableObject + @Published
Adopted by a class. When a @Published property changes, subscribed Views re-render
@StateObject — Owner
Creates and owns the object. The object persists even when the View is recreated. Used in the top-level View
@ObservedObject — Subscriber
Receives and subscribes to an externally injected object. May be replaced with a new object when the View is recreated
Common Mistake
Using @StateObject in a child View causes the object to reset when the parent re-renders
Topic 04

@Observable · @EnvironmentObject

iOS 17+ new approach and environment injection
Swift 
// iOS 17+ approach
@Observable
class UserSettings {
    var username = "DevRi"
    var isDarkMode = false
}

struct AppRoot: View {
    @State var settings = UserSettings()

    var body: some View {
        SettingsView()
            .environment(settings)
    }
}

struct SettingsView: View {
    @Environment(UserSettings.self)
    var settings

    var body: some View {
        Text(settings.username)
    }
}
Think About It
  • Why does using the @Observable macro (iOS 17+) eliminate the need for @Published?
  • What problem does @EnvironmentObject solve? (What is prop drilling?)
  • Where is \.dismiss defined in @Environment(\.dismiss)?
@Observable macro (iOS 17+)
Attach to a class and it works without ObservableObject + @Published. Much simpler
@EnvironmentObject
Inject from parent via .environmentObject(). Deep children can access it directly. Solves prop drilling
@Environment(\.store)
@Observable + @Environment combo (iOS 17+). A more explicit and type-safe approach

Think About It
  • Why should you put a NavigationStack inside each tab of a TabView?
  • When would you need programmatic tab switching with TabView(selection: $selectedTab)?

.tabItem { Label("Home", systemImage: "house") }
Icon and label for each tab. Uses SF Symbols names
Independent NavigationStack per tab
Placing a NavigationStack inside each tab preserves each tab's navigation state when switching tabs
Programmatic tab switching
TabView(selection: $selectedTab). Navigate to a specific tab from push notifications, etc.

2. Shared Data Model (@Observable)
Swift 
@Observable
class AppData {
    var favorites: [String] = []

    func toggle(_ item: String) {
        if favorites.contains(item) {
            favorites.removeAll { $0 == item }
        } else {
            favorites.append(item)
        }
    }
}

struct AppRoot: View {
    @State var data = AppData()

    var body: some View {
        TabView {
            ItemListView()
                .tabItem { Label("List",
                    systemImage: "list.bullet") }
            FavoritesView()
                .tabItem { Label("Favorites",
                    systemImage: "heart") }
        }
        .environment(data)
    }
}

struct ItemListView: View {
    @Environment(AppData.self) var data
    let items = ["Apple", "Banana", "Grape"]

    var body: some View {
        List(items, id: \.self) { item in
            HStack {
                Text(item)
                Spacer()
                Image(systemName:
                    data.favorites.contains(item)
                    ? "heart.fill" : "heart")
                    .foregroundStyle(.red)
                    .onTapGesture {
                        data.toggle(item)
                    }
            }
        }
    }
}

struct FavoritesView: View {
    @Environment(AppData.self) var data

    var body: some View {
        List(data.favorites, id: \.self) {
            Text($0)
        }
    }
}
3. Memo App
Swift 
struct Memo: Identifiable {
    let id = UUID()
    var title: String
    var body: String
    var date = Date()
}

class MemoStore: ObservableObject {
    @Published var memos: [Memo] = [
        Memo(title: "First Memo",
             body: "Starting SwiftUI study!")
    ]
}

struct MemoListView: View {
    @StateObject var store = MemoStore()
    @State private var showNew = false

    var body: some View {
        NavigationStack {
            List {
                ForEach(store.memos) { memo in
                    NavigationLink {
                        MemoDetailView(memo: memo)
                    } label: {
                        VStack(alignment: .leading) {
                            Text(memo.title)
                                .font(.headline)
                            Text(memo.body)
                                .font(.caption)
                                .foregroundStyle(
                                    .secondary)
                                .lineLimit(1)
                        }
                    }
                }
            }
            .navigationTitle("Memos")
            .toolbar {
                Button { showNew = true } label: {
                    Image(systemName: "plus")
                }
            }
            .sheet(isPresented: $showNew) {
                NewMemoView(store: store)
            }
        }
    }
}

struct MemoDetailView: View {
    let memo: Memo
    var body: some View {
        ScrollView {
            Text(memo.body)
                .padding()
        }
        .navigationTitle(memo.title)
    }
}

struct NewMemoView: View {
    @ObservedObject var store: MemoStore
    @Environment(\.dismiss) var dismiss
    @State private var title = ""
    @State private var body = ""

    var body_view: some View {
        NavigationStack {
            Form {
                TextField("Title", text: $title)
                TextEditor(text: $body)
                    .frame(minHeight: 120)
            }
            .navigationTitle("New Memo")
            .toolbar {
                Button("Save") {
                    store.memos.append(
                        Memo(title: title,
                             body: body))
                    dismiss()
                }
            }
        }
    }
}
