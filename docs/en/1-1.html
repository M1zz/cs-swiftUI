Think About It
  • Does .frame(width: 200) change the view's own size, or the size of a frame wrapping the view?
  • When do you use maxWidth: .infinity? How do you make a button span the full screen width?
  • What does the alignment parameter inside .frame() align?
.frame(width:height:)

maxWidth: .infinity
Takes up all the width the parent offers. The full-width button pattern
alignment parameter

1. Build a self-introduction card
Swift 
struct ProfileCard: View {
    var body: some View {
        ZStack {
            RoundedRectangle(cornerRadius: 20)
                .fill(.orange.opacity(0.1))

            VStack(spacing: 12) {
                Image(systemName: "person.circle.fill")
                    .font(.system(size: 60))
                    .foregroundStyle(.orange)

                Text("John Doe")
                    .font(.title2)
                    .bold()

                HStack {
                    Image(systemName: "swift")
                    Text("Aspiring iOS Developer")
                        .font(.subheadline)
                        .foregroundStyle(.secondary)
                }
            }
            .padding(24)
        }
        .frame(maxWidth: .infinity)
        .padding()
    }
}
2. Modifier order experiment
Swift 
VStack(spacing: 30) {
    // padding first → background covers the padded area
    Text("padding → background")
        .padding()
        .background(.orange.opacity(0.3))
        .cornerRadius(8)

    // background first → background covers only the text size
    Text("background → padding")
        .background(.blue.opacity(0.3))
        .padding()
        .cornerRadius(8)
}
// Modifiers are applied from inside → outside.
// When .padding() comes first, spacing is added then background is painted,
// When .background() comes first, background is painted on the text size only.
3. Weather card UI clone
Swift 
struct WeatherCard: View {
    var body: some View {
        GeometryReader { geo in
            ZStack {
                LinearGradient(
                    colors: [.blue, .cyan],
                    startPoint: .top,
                    endPoint: .bottom
                )
                .cornerRadius(20)

                VStack(spacing: 8) {
                    Text("Seoul")
                        .font(.title2).bold()
                        .foregroundStyle(.white)
                    Image(systemName: "sun.max.fill")
                        .font(.system(size: 48))
                        .foregroundStyle(.yellow)
                    Text("24°")
                        .font(.system(size: 56,
                            weight: .thin))
                        .foregroundStyle(.white)
                }
            }
            .frame(width: geo.size.width * 0.9,
                   height: 200)
            .frame(maxWidth: .infinity)
        }
        .frame(height: 200)
        .padding()
    }
}
