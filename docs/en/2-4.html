Topic 01

withAnimation & .animation() — Smooth State Changes

Wrap @State changes in animation and SwiftUI interpolates between before and after for smooth transitions
Swift
struct AnimationDemo: View {
    @State private var isExpanded = false
    @State private var rotation: Double = 0
    @State private var scale: CGFloat = 1.0

    var body: some View {
        VStack(spacing: 32) {

            // withAnimation — animates all state changes in the closure
            RoundedRectangle(cornerRadius: 16)
                .fill(Color.orange)
                .frame(
                    width: isExpanded ? 280 : 120,
                    height: isExpanded ? 160 : 60
                )
                .onTapGesture {
                    withAnimation(.spring(response: 0.4,
                                         dampingFraction: 0.7)) {
                        isExpanded.toggle()
                    }
                }

            // .animation() view modifier — when a specific value changes
            Image(systemName: "star.fill")
                .font(.system(size: 40))
                .foregroundStyle(.orange)
                .rotationEffect(.degrees(rotation))
                .scaleEffect(scale)
                .animation(.easeInOut(duration: 0.3), value: rotation)
                .animation(.spring(), value: scale)
                .onTapGesture {
                    rotation += 45
                    scale = scale == 1.0 ? 1.5 : 1.0
                }

            // Various animation curves
            Button("easeIn") {
                withAnimation(.easeIn(duration: 0.5)) { isExpanded.toggle() }
            }
            Button("spring") {
                withAnimation(.spring(
                    response: 0.5,
                    dampingFraction: 0.6,
                    blendDuration: 0
                )) { isExpanded.toggle() }
            }
            Button("delay + repeat") {
                withAnimation(
                    .easeInOut(duration: 0.3)
                    .delay(0.1)
                    .repeatCount(3, autoreverses: true)
                ) { scale = 1.3 }
            }
        }
    }
}
Think About It
  • What is the difference in scope between withAnimation and .animation(value:)?
  • In .spring(response:dampingFraction:), what happens as dampingFraction approaches 1.0?
  • How do you get a callback when an animation completes? (withAnimation(completion:) iOS 17+)
withAnimation
Wraps all @State changes within the closure as animated. Ideal for animating multiple states simultaneously
.animation(value:)
Applies animation only to this view when a specific value changes. Finer-grained control. Value must be Equatable
Animation Curves
.easeIn/.easeOut/.easeInOut: consistent acceleration/deceleration | .spring: physics-based bounce | .linear: constant speed
Topic 02

.transition() — View Appear/Disappear Animations

Defines the animation when a View is added to or removed from the view hierarchy
Swift
struct TransitionDemo: View {
    @State private var showBanner = false
    @State private var showCard = false
    @State private var showModal = false

    var body: some View {
        VStack(spacing: 20) {

            // Basic transition
            Button("Toggle Banner") {
                withAnimation(.spring()) {
                    showBanner.toggle()
                }
            }
            if showBanner {
                Text("🎉 Sale event in progress!")
                    .padding()
                    .background(Color.orange.opacity(0.15))
                    .cornerRadius(12)
                    .transition(.slide)       // slides in from the side
            }

            // .scale + .opacity combined
            Button("Toggle Card") {
                withAnimation(.easeOut(duration: 0.3)) {
                    showCard.toggle()
                }
            }
            if showCard {
                RoundedRectangle(cornerRadius: 16)
                    .fill(Color.orange.opacity(0.2))
                    .frame(height: 100)
                    .transition(.scale.combined(with: .opacity))
            }

            // asymmetric — different enter/exit
            Button("Toggle Modal") {
                withAnimation(.spring(response: 0.45)) {
                    showModal.toggle()
                }
            }
            if showModal {
                Text("Slides up from bottom, exits to top")
                    .padding()
                    .background(Color.orange)
                    .foregroundStyle(.white)
                    .cornerRadius(12)
                    .transition(.asymmetric(
                        insertion: .move(edge: .bottom)
                            .combined(with: .opacity),
                        removal: .move(edge: .top)
                            .combined(with: .opacity)
                    ))
            }
        }
        .padding()
    }
}

// AnyTransition extension — custom transition
extension AnyTransition {
    static var popUp: AnyTransition {
        .scale(scale: 0.1, anchor: .bottom)
        .combined(with: .opacity)
    }
}
Think About It
  • Why does .transition() require withAnimation to work?
  • What is the difference between .move(edge: .bottom) and .slide?
  • How do you apply transitions to item additions/removals in a List?
Built-in Transitions
.opacity · .scale · .slide · .move(edge:) · .push(from:) — can be combined
.combined(with:)
Apply two transitions simultaneously. Chain like .scale.combined(with: .opacity)
.asymmetric
Apply different transitions for insertion and removal. Creates directional UX
Topic 03

matchedGeometryEffect — Hero Animations

Create continuity between two views — like a list item naturally expanding into a detail card
Swift
struct HeroAnimationView: View {
    @Namespace private var heroNS
    @State private var selectedCard: Int? = nil

    let cards = [1, 2, 3]

    var body: some View {
        ZStack {
            // List — small cards
            if selectedCard == nil {
                VStack(spacing: 12) {
                    ForEach(cards, id: \.self) { id in
                        RoundedRectangle(cornerRadius: 16)
                            .fill(Color.orange.opacity(0.15))
                            .frame(height: 80)
                            .overlay(
                                Text("Card \(id)")
                                    .font(.headline)
                            )
                            // Assign heroNS + "card\(id)" to this view
                            .matchedGeometryEffect(
                                id: "card\(id)",
                                in: heroNS
                            )
                            .onTapGesture {
                                withAnimation(.spring(
                                    response: 0.45,
                                    dampingFraction: 0.8
                                )) {
                                    selectedCard = id
                                }
                            }
                    }
                }
                .padding()
            }

            // Detail — expanded card
            if let id = selectedCard {
                RoundedRectangle(cornerRadius: 24)
                    .fill(Color.orange.opacity(0.25))
                    .ignoresSafeArea()
                    .overlay(
                        VStack {
                            Text("Card \(id) Detail")
                                .font(.largeTitle.bold())
                            Button("Close") {
                                withAnimation(.spring(
                                    response: 0.45,
                                    dampingFraction: 0.8
                                )) {
                                    selectedCard = nil
                                }
                            }
                            .padding(.top)
                        }
                    )
                    // Same id + namespace → SwiftUI interpolates
                    .matchedGeometryEffect(
                        id: "card\(id)",
                        in: heroNS
                    )
            }
        }
    }
}

// @Namespace: an ID space shared between views
// matchedGeometryEffect interpolates position/size when
// views with the same namespace + id are swapped
Think About It
  • How does matchedGeometryEffect behave if both views are on screen simultaneously? (isSource parameter)
  • Can you create multiple @Namespaces? When should you separate them?
  • What is the new iOS 18 API for applying hero animations to NavigationStack page transitions?
@Namespace
A unique ID space shared between views. Views with the same namespace + id are connected by animation
Interpolated Elements
Automatically interpolates position, size, and cornerRadius. Adjustable via the properties parameter
ZStack + if/else Pattern
Swap list view and detail view with if/else inside a ZStack. Both views should not render simultaneously for smooth results
Topic 04

Custom ViewModifier — Reusable Styles

Encapsulate repeated modifier combinations into a single type — the UI version of Protocol + Extension
Swift
// Card style modifier
struct CardModifier: ViewModifier {
    var color: Color = .orange

    func body(content: Content) -> some View {
        content
            .padding(20)
            .background(color.opacity(0.1))
            .overlay(
                RoundedRectangle(cornerRadius: 16)
                    .stroke(color.opacity(0.3), lineWidth: 1)
            )
            .cornerRadius(16)
            .shadow(color: color.opacity(0.15),
                    radius: 8, x: 0, y: 4)
    }
}

// Shimmer loading effect modifier
struct ShimmerModifier: ViewModifier {
    @State private var phase: CGFloat = 0

    func body(content: Content) -> some View {
        content
            .overlay(
                LinearGradient(
                    gradient: Gradient(colors: [
                        .clear,
                        .white.opacity(0.5),
                        .clear
                    ]),
                    startPoint: .init(x: phase - 0.3, y: 0),
                    endPoint: .init(x: phase + 0.3, y: 0)
                )
                .blendMode(.screen)
            )
            .onAppear {
                withAnimation(
                    .linear(duration: 1.5)
                    .repeatForever(autoreverses: false)
                ) {
                    phase = 1.3
                }
            }
    }
}

// Pressable effect modifier
struct PressableModifier: ViewModifier {
    @State private var isPressed = false

    func body(content: Content) -> some View {
        content
            .scaleEffect(isPressed ? 0.95 : 1.0)
            .opacity(isPressed ? 0.8 : 1.0)
            .animation(.easeOut(duration: 0.1), value: isPressed)
            .simultaneousGesture(
                DragGesture(minimumDistance: 0)
                    .onChanged { _ in isPressed = true }
                    .onEnded   { _ in isPressed = false }
            )
    }
}

// View Extension for convenient usage
extension View {
    func cardStyle(color: Color = .orange) -> some View {
        modifier(CardModifier(color: color))
    }
    func shimmer() -> some View {
        modifier(ShimmerModifier())
    }
    func pressable() -> some View {
        modifier(PressableModifier())
    }
}

// Usage
Text("Hello").cardStyle()
Rectangle().shimmer()
Button("Tap!") { }.pressable()
Think About It
  • What is the difference between a ViewModifier and a custom View? When do you choose which?
  • Why does ShimmerModifier have @State? Does each view instance have its own independent state?
  • Does the order matter when chaining multiple ViewModifiers?
ViewModifier Protocol
Implement body(content:). content is the existing view. Add modifiers and return a new view
View Extension Pattern
extension View { func myStyle() → some View } makes it chainable like built-in SwiftUI modifiers
@State in Modifier
ViewModifiers can have @State. Each view the modifier is applied to gets its own independent state instance
Topic 05

Gesture — Tap · Drag · Swipe

Detect user finger movements and build interactive UI that responds to them
Swift
struct GestureDemo: View {
    @State private var offset: CGSize = .zero
    @State private var isDragging = false
    @State private var tapCount = 0
    @State private var isLongPressed = false

    var body: some View {
        VStack(spacing: 40) {

            // TapGesture
            Text("Tap count: \(tapCount)")
                .padding()
                .background(Color.orange.opacity(0.15))
                .cornerRadius(12)
                .onTapGesture(count: 2) {   // double tap
                    tapCount += 1
                }

            // DragGesture — drag to move
            Circle()
                .fill(isDragging ? Color.orange : Color.orange.opacity(0.5))
                .frame(width: 80, height: 80)
                .offset(offset)
                .scaleEffect(isDragging ? 1.1 : 1.0)
                .animation(.spring(), value: isDragging)
                .gesture(
                    DragGesture()
                        .onChanged { value in
                            offset = value.translation
                            isDragging = true
                        }
                        .onEnded { _ in
                            withAnimation(.spring()) {
                                offset = .zero  // snap back
                                isDragging = false
                            }
                        }
                )

            // LongPressGesture
            RoundedRectangle(cornerRadius: 16)
                .fill(isLongPressed ? Color.orange : Color.gray.opacity(0.2))
                .frame(height: 60)
                .overlay(
                    Text(isLongPressed ? "Long pressing..." : "Long press me")
                )
                .gesture(
                    LongPressGesture(minimumDuration: 0.5)
                        .onChanged { _ in
                            withAnimation { isLongPressed = true }
                        }
                        .onEnded { _ in
                            withAnimation { isLongPressed = false }
                        }
                )

            // simultaneousGesture — recognize multiple gestures at once
            Text("Simultaneous tap + drag")
                .onTapGesture { print("Tap!") }
                .simultaneousGesture(
                    DragGesture().onChanged { _ in print("Drag!") }
                )
        }
        .padding()
    }
}
Think About It
  • What is the difference between .gesture() and .onTapGesture()? How is gesture priority determined?
  • What is the difference between DragGesture's value.translation and value.predictedEndTranslation?
  • If you were to implement swipe-to-delete yourself, which gesture would you use?
Key Gesture Types
TapGesture · DragGesture · LongPressGesture · MagnificationGesture · RotationGesture
onChanged / onEnded
onChanged: called every frame during the gesture | onEnded: called once when the finger lifts
simultaneousGesture
Recognizes parent-child view gestures without conflict. Use .highPriorityGesture for priority control

Path & Shape — Custom Shapes

Draw things that basic shapes can't express — waves, progress indicators, speech bubbles, and more
Swift
// Custom Shape — adopts the Shape protocol
struct Wave: Shape {
    var amplitude: CGFloat = 10
    var frequency: CGFloat = 2

    func path(in rect: CGRect) -> Path {
        var path = Path()
        path.move(to: CGPoint(x: 0, y: rect.midY))

        let wavelength = rect.width / frequency
        for x in stride(from: 0, to: rect.width, by: 1) {
            let y = rect.midY + amplitude *
                sin(2 * .pi * x / wavelength)
            path.addLine(to: CGPoint(x: x, y: y))
        }
        return path
    }
}

// Circular progress indicator
struct ArcProgressShape: Shape {
    var progress: Double   // 0.0 ~ 1.0
    var animatableData: Double {
        get { progress }
        set { progress = newValue }
    }

    func path(in rect: CGRect) -> Path {
        var path = Path()
        let center = CGPoint(x: rect.midX, y: rect.midY)
        let radius = min(rect.width, rect.height) / 2 - 8

        path.addArc(
            center: center,
            radius: radius,
            startAngle: .degrees(-90),
            endAngle: .degrees(-90 + 360 * progress),
            clockwise: false
        )
        return path
    }
}

// Actual usage
struct ProgressRing: View {
    @State private var progress = 0.0

    var body: some View {
        ZStack {
            Circle()
                .stroke(Color.orange.opacity(0.2),
                        lineWidth: 12)
            ArcProgressShape(progress: progress)
                .stroke(
                    Color.orange,
                    style: StrokeStyle(
                        lineWidth: 12,
                        lineCap: .round
                    )
                )
                .animation(.easeInOut(duration: 1.5),
                           value: progress)
            Text("\(Int(progress * 100))%")
                .font(.title2.bold())
        }
        .frame(width: 140, height: 140)
        .onAppear { progress = 0.75 }
    }
}
Think About It
  • When you implement animatableData, which values of a Shape can be animated?
  • In Path.addArc, how does the SwiftUI coordinate system's direction differ from mathematics?
  • How does Shape differ from Canvas? Is there a performance difference for complex graphics?
Shape Protocol
Implement path(in rect:). rect is the space the view occupies. Draw the desired form with Path and return it
animatableData
Specifies which property of the Shape to animate. SwiftUI interpolates from start to end values, calling it each frame
StrokeStyle
Fine-tune line styles: lineWidth, lineCap(.round/.square), dash pattern, etc. Use .stroke(style:) instead of .stroke()
